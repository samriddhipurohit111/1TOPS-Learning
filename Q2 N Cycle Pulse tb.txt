module tb_pulse_stretcher;

    reg clk, rst;
    reg pulse_in;
    wire pulse_out;

    pulse_stretcher DUT (
        .clk(clk),
        .rst(rst),
        .pulse_in(pulse_in),
        .pulse_out(pulse_out)
    );

    always #5 clk = ~clk;

    initial begin
        clk = 0;
        rst = 1;
        pulse_in = 0;

        #10 rst = 0;

        // 1-cycle pulse
        #10 pulse_in = 1;
        #10 pulse_in = 0;

        #100 $finish;
    end

endmodule
