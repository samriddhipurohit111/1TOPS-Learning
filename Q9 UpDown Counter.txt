`timescale 1ns / 1ps

module FourBitUPOrDown(
    input clk,
    input rst,
    input dir,
    output reg [3:0] count
    );
    
    always@(posedge clk)begin
    if(rst)begin
    count<=4'b0000;
    end
    else begin
    if(dir)
    count<=count+1;
    else 
    count<=count-1;
    end
    
    end
endmodule
