`timescale 1ns / 1ps


module falling_edge_pulse (
    input  clk,
    input  reset,
    input  in,
    output reg  pulse
);

    reg prevIn;

    always @(posedge clk) begin
        if (reset) begin
            prevIn <= 1'b0;
            pulse   <= 1'b0;
        end
        else begin
            pulse   <= prevIn & ~in; // detect 1 â†’ 0 transition
            prevIn <= in;             // store current input
        end
    end

endmodule
