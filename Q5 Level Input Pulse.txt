module RisingEdgeLevel(
    input  wire clk,
    input  wire rst,
    input  wire sig_in,
    output reg  pulse_out
);

    reg sig_d;   // stores previous value of sig_in

    always @(posedge clk) begin
        if (rst) begin
            sig_d      <= 1'b0;
            pulse_out <= 1'b0;
        end
        else begin
            sig_d      <= sig_in;              // save current input
            pulse_out <= sig_in & ~sig_d;      // rising edge detect
        end
    end

endmodule
