`timescale 1ns / 1ps

module first_event_capture (
    input  wire clk,
    input  wire reset,
    input  wire eventIn,
    output reg  firstEvent
);

 reg captured;
 always@(posedge clk)begin 
 if(reset)begin 
 captured<=1'b0;
 firstEvent<=1'b0;
 
 end
 
 else if (eventIn&& !captured) begin
 captured<=1'b1;
 firstEvent<=1'b1;
 
 
 
 end
 end
 endmodule
