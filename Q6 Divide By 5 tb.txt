`timescale 1ns/1ps

module tb_DivideBy5;

    reg  clk;
    reg  reset;
    wire clk_div5;

    // Instantiate DUT
    DivideBy5 dut (
        .clk(clk),
        .reset(reset),
        .clk_div5(clk_div5)
    );

    // Generate input clock (10 ns period = 100 MHz)
    initial begin
        clk = 0;
        forever #5 clk = ~clk;   // toggles every 5 ns
    end

    // Apply reset
    initial begin
        reset = 1'b1;
        #20;
        reset = 1'b0;
    end

    // Stop simulation
    initial begin
        #500 $finish;
    end

endmodule
