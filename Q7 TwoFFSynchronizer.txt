`timescale 1ns / 1ps


module TwoFFSync(
    input clk_dst,
    input rst_dst,
    input async_in,
    output reg sync_out
    );
    reg sync_ff1;
    always @(posedge clk_dst)begin
    if(rst_dst)begin
     sync_ff1<=1'b0;
    sync_out<=1'b0;
   
    end
    else begin
    sync_ff1<=async_in;
    sync_out<=sync_ff1;
    end
    end
    
endmodule
